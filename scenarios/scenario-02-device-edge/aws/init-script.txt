#!/bin/bash

set -eo pipefail

# Prerequisites
sudo dnf install -y wget

# Install Flightctl
sudo dnf config-manager addrepo --from-repofile=https://rpm.flightctl.io/flightctl-fedora.repo
sudo dnf install -y flightctl-services-1.1.0~rc4-1.fc43.x86_64
sudo dnf install -y flightctl-cli-1.1.0~rc4-1.fc43.x86_64
sudo dnf install -y flightctl-observability-1.1.0~rc4-1.fc43.x86_64

# Install yq
VERSION=v4.52.2
PLATFORM=linux_amd64

wget https://github.com/mikefarah/yq/releases/download/${VERSION}/yq_${PLATFORM}.tar.gz -O - | tar xz && sudo mv yq_${PLATFORM} /usr/local/bin/yq

# Set global.baseDomain
FQDN=$(curl ifconfig.me).nip.io yq -i '.global.baseDomain = strenv(FQDN)' /etc/flightctl/service-config.yaml

# Start Flightctl service
sudo systemctl enable --now flightctl.target

# Add user to PAM issuer
export USER=mlops
export PASSWORD=mlops123

sudo podman exec -i flightctl-pam-issuer groupadd flightctl-admin 2>/dev/null || true && \
sudo podman exec flightctl-pam-issuer adduser $USER && \
sudo podman exec -i flightctl-pam-issuer sh -c "echo \"$USER:$PASSWORD\" | chpasswd" && \
sudo podman exec -i flightctl-pam-issuer usermod -aG flightctl-admin $USER

# Login to Flightctl 
sudo flightctl login -k https://$(curl ifconfig.me).nip.io:3443 -u mlops -p mlops123

# Start observability service
sudo systemctl enable --now flightctl-observability.target

# Download sample Grafana dashboards provided by the vLLM project
curl https://raw.githubusercontent.com/vllm-project/vllm/refs/heads/main/examples/online_serving/dashboards/grafana/performance_statistics.json --output-dir /etc/flightctl/flightctl-grafana/provisioning/dashboards/flightctl -O
curl https://raw.githubusercontent.com/vllm-project/vllm/refs/heads/main/examples/online_serving/dashboards/grafana/query_statistics.json --output-dir /etc/flightctl/flightctl-grafana/provisioning/dashboards/flightctl -O

# Prepare script used to configure the Grafana datasources

cat > /etc/flightctl/flightctl-grafana/provisioning/datasources/mlops-fleet.yaml << EOF
apiVersion: 1
datasources: []
EOF

cat > /opt/set-grafana-datasources.sh << 'EOF'
#! /bin/bash
FLIGHTCTL_USERNAME=mlops
FLIGHTCTL_PASSWORD=mlops123
FLIGHTCTL_FLEET=mlops

sudo flightctl login -k https://$(curl ifconfig.me).nip.io:3443 -u $FLIGHTCTL_USERNAME -p $FLIGHTCTL_PASSWORD

device_names=$(sudo flightctl get device --field-selector=metadata.owner=Fleet/"$FLIGHTCTL_FLEET" -o name)
for device_name in $device_names
do
    ip=$(sudo flightctl console device/$device_name -- curl -s http://169.254.169.254/latest/meta-data/public-ipv4)
    alias=$(sudo flightctl get device $device_name -o yaml | yq '.metadata.labels.alias')
    sudo NAME=$alias IP=$ip yq -i '.datasources += [{"name": strenv(NAME), "type": "prometheus", "url": "http://" + strenv(IP) + ".nip.io:9090"}]' /etc/flightctl/flightctl-grafana/provisioning/datasources/mlops-fleet.yaml
done

sudo systemctl restart flightctl-observability.target
EOF

sudo chmod +x /opt/set-grafana-datasources.sh
